# BossCrafts Automated Docker Cleaner

# "Three minds. One forge. Infinite creations."



param(

    [string]$Age = "24h",          # How old containers must be before pruning

    [switch]$PruneVolumes,         # Optional: prune unused volumes

    [switch]$Silent                # Optional: suppress console output

)



$timestamp = (Get-Date).ToString("yyyy-MM-dd_HH-mm-ss")

$logDir = "$HOME\docker-cleaner-logs"

$logFile = "$logDir\cleanup_$timestamp.log"



if (!(Test-Path $logDir)) {

    New-Item -ItemType Directory -Path $logDir | Out-Null

}



function Log {

    param([string]$msg)

    $msg | Tee-Object -FilePath $logFile -Append | Out-Null

}



Log "=== BossCrafts Docker Cleanup Ritual Started at $timestamp ==="



# Check Docker availability

try {

    docker info | Out-Null

} catch {

    Log "Docker is not running or not installed. Ritual aborted."

    exit 1

}



Log "Docker detected. Beginning cleanup..."



# Step 1: prune old containers

Log "Pruning stopped containers older than $Age..."

docker container prune --force --filter "until=$Age" | Tee-Object -FilePath $logFile -Append



# Step 2: prune unused images, networks, and build cache

Log "Pruning unused images, networks, and dangling build cache..."

docker system prune --force | Tee-Object -FilePath $logFile -Append



# Step 3: optional volume pruning

if ($PruneVolumes) {

    Log "Pruning unused volumes (dangerous)..."

    docker system prune --volumes --force | Tee-Object -FilePath $logFile -Append

} else {

    Log "Skipping volume pruning."

}



Log "Cleanup ritual complete."

Log "Log saved to: $logFile"



if (-not $Silent) {

    Write-Host "Docker cleanup complete. Log saved to $logFile"

}

# BossCrafts Hearth‑Tender Daemon

# "Three minds. One forge. Infinite creations."



param(

    [int]$CheckIntervalMinutes = 10,

    [int]$WarnThreshold = 70,

    [int]$FullThreshold = 85,

    [int]$EmergencyThreshold = 95,

    [int]$WSLMaxGB = 60,

    [string]$AgentEventPath = "$HOME\BossCrafts\events",

    [switch]$AllowVHDXCompaction,

    [switch]$AllowVolumePrune

)



# Ensure event directory exists

if (!(Test-Path $AgentEventPath)) {

    New-Item -ItemType Directory -Path $AgentEventPath | Out-Null

}



function Emit-Rune {

    param($event, $level, $usage, $action)



    $payload = @{

        event     = $event

        level     = $level

        usage     = $usage

        action    = $action

        timestamp = (Get-Date).ToString("o")

    } | ConvertTo-Json -Depth 5



    $file = Join-Path $AgentEventPath "rune_$((Get-Date).ToString('yyyyMMdd_HHmmss')).json"

    $payload | Out-File $file -Encoding utf8

}



function Get-DockerUsage {

    $info = docker system df --format "{{json .}}" | ConvertFrom-Json

    return [int]($info.DiskUsedPercent)

}



function Compact-WSL {

    param($maxGB)



    $vhdx = "$env:LOCALAPPDATA\Docker\wsl\data\ext4.vhdx"

    if (!(Test-Path $vhdx)) { return }



    $sizeGB = (Get-Item $vhdx).Length / 1GB



    if ($sizeGB -gt $maxGB) {

        wsl --shutdown

        Optimize-VHD -Path $vhdx -Mode Full

        Emit-Rune "wsl_compaction" "info" "$sizeGB GB" "compact_vhdx"

    }

}



Write-Host "Hearth‑Tender Daemon awakened..."



while ($true) {



    try {

        $usage = Get-DockerUsage

    } catch {

        Start-Sleep -Seconds ($CheckIntervalMinutes * 60)

        continue

    }



    if ($usage -ge $EmergencyThreshold) {

        docker system prune --force

        Emit-Rune "docker_cleanup" "critical" "$usage%" "emergency_prune"

    }

    elseif ($usage -ge $FullThreshold) {

        docker system prune --force

        if ($AllowVolumePrune) {

            docker system prune --volumes --force

        }

        Emit-Rune "docker_cleanup" "warning" "$usage%" "full_prune"

    }

    elseif ($usage -ge $WarnThreshold) {

        docker container prune --force --filter "until=24h"

        docker image prune --force

        Emit-Rune "docker_cleanup" "notice" "$usage%" "light_prune"

    }



    if ($AllowVHDXCompaction) {

        Compact-WSL -maxGB $WSLMaxGB

    }



    Start-Sleep -Seconds ($CheckIntervalMinutes * 60)

}

def load_events(limit=200):

    events = []

    if not os.path.isdir(EVENTS_DIR):

        return events



    files = sorted(

        [os.path.join(EVENTS_DIR, f) for f in os.listdir(EVENTS_DIR) if f.endswith(".json")],

        reverse=True

    )



    for path in files[:limit]:

        try:

            with open(path, "r", encoding="utf-8") as f:

                data = json.load(f)

            if data.get("type") != "event":

                continue

            events.append({

                "timestamp": data.get("timestamp"),

                "source": data.get("source"),

                "event": data.get("event"),

                "level": data.get("level", "info"),

                "usage": data.get("usage"),

                "action": data.get("action"),

                "meta": data.get("meta"),

            })

        except Exception:

            continue

    return events



@app.route("/", methods=["GET"])

def index():

    filters = {

        "source": request.args.get("source") or None,

        "level": request.args.get("level") or None,

        "event": request.args.get("event") or None,

    }



    events = load_events()

    def match(e):

        if filters["source"] and filters["source"].lower() not in (e["source"] or "").lower():

            return False

        if filters["level"] and filters["level"].lower() not in (e["level"] or "").lower():

            return False

        if filters["event"] and filters["event"].lower() not in (e["event"] or "").lower():

            return False

        return True



    events = [e for e in events if match(e)]

    return render_template_string(TEMPLATE, events=events, filters=filters)



@app.route("/command", methods=["POST"])

def command():

    target = request.form.get("target")

    command = request.form.get("command")

    args_raw = request.form.get("args") or "{}"

    issued_by = request.form.get("issued_by") or "boss"



    try:

        args = json.loads(args_raw)

    except json.JSONDecodeError:

        args = {"raw": args_raw}



    payload = {

        "type": "command",

        "target": target,

        "command": command,

        "args": args,

        "issued_by": issued_by,

    path = os.path.join(COMMANDS_DIR, fname)

    with open(path, "w", encoding="utf-8") as f:

        json.dump(payload, f, indent=2)



    return redirect("/")

import os, json, argparse, time

from datetime import datetime, timezone



ROOT = os.path.expanduser("~/BossCrafts")

BUS = os.path.join(ROOT, "bus")

EVENTS = os.path.join(BUS, "events")

COMMANDS = os.path.join(BUS, "commands")



AGENTS = {

    "hearth": "hearth_tender",

    "archivist": "archivist",

    "codemage": "codemage",

    "runeforge": "runeforge",

    "devlot": "devlot"

}



def emit_command(target, command, args):

    payload = {

        "type": "command",

        "target": target,

        "command": command,

        "args": args,

        "issued_by": "cli",

        "timestamp": datetime.utcnow().isoformat() + "Z"

    }

    fname = f"cmd_{datetime.utcnow().strftime('%Y%m%d_%H%M%S_%f')}.json"

    path = os.path.join(COMMANDS, fname)

    with open(path, "w", encoding="utf-8") as f:

        json.dump(payload, f, indent=2)

    print(f"✔ Command sent → {target}:{command}")



def tail_events():

    print("=== Tailing BossCrafts Events ===")

    seen = set()

    while True:

        files = sorted(os.listdir(EVENTS))

        for f in files:

            if f in seen: continue

            seen.add(f)

            try:

                with open(os.path.join(EVENTS, f), "r", encoding="utf-8") as fp:

                    data = json.load(fp)

                ts = data.get("timestamp")

                src = data.get("source")

                evt = data.get("event")

                lvl = data.get("level")

                act = data.get("action")

                print(f"[{ts}] {lvl.upper():8} {src:14} → {evt} ({act})")

            except:

                pass

        time.sleep(1)



def show_status():

    now = datetime.now(timezone.utc)

    latest = {a: None for a in AGENTS.values()}



    files = sorted(os.listdir(EVENTS), reverse=True)

    for f in files:

        try:

            with open(os.path.join(EVENTS, f), "r", encoding="utf-8") as fp:

                data = json.load(fp)

            if data.get("event") == "heartbeat":

                src = data.get("source")

                if src in latest:

                    ts = datetime.fromisoformat(data["timestamp"].replace("Z","+00:00"))

                    if latest[src] is None or ts > latest[src]:

                        latest[src] = ts

        except:

            continue



    print("=== BossCrafts Agent Status ===")

    for agent, ts in latest.items():

        if ts is None:

            print(f"{agent:14} → OFFLINE (no heartbeat)")

        else:

            age = (now - ts).total_seconds()

            if age < 30:

                state = "ONLINE"

            elif age < 120:

                state = "LAGGING"

            else:

                state = "STALE"

            print(f"{agent:14} → {state:7} ({int(age)}s ago)")



def main():

    parser = argparse.ArgumentParser(description="BossCrafts CLI")

    parser.add_argument("agent", nargs="?", help="agent or command")

    parser.add_argument("command", nargs="?", help="command")

    parser.add_argument("--args", default="{}", help="JSON args")

    parser.add_argument("--tail", action="store_true", help="tail events")

    parser.add_argument("--status", action="store_true", help="show status")



    a = parser.parse_args()



    if a.status:

        show_status()

        return



    if a.tail:

        tail_events()

        return



    if not a.agent:

        print("Usage: bforge <agent> <command> [--args '{}']")

        return



    if a.agent in ["status"]:

        show_status()

        return



    if a.agent in ["events", "tail"]:

        tail_events()

        return



    # agent command

    if a.agent not in AGENTS:

        print(f"Unknown agent: {a.agent}")

        return



    target = AGENTS[a.agent]

    command = a.command

    try:

        args = json.loads(a.args)

    except:

        args = {"raw": a.args}



    emit_command(target, command, args)



if __name__ == "__main__":

    main()

import psutil  # you’ll add this to the venv



def snapshot_disk():

    root = os.path.splitdrive(os.path.expanduser("~"))[0] + "\\"

    usage = psutil.disk_usage(root)

    return {

        "root": root,

        "total_gb": round(usage.total / (1024**3), 1),

        "used_gb": round(usage.used / (1024**3), 1),

        "free_gb": round(usage.free / (1024**3), 1),

        "percent": usage.percent,

    }



def snapshot_docker():

    try:

        import subprocess, json as _json

        out = subprocess.check_output(["docker", "system", "df", "--format", "{{json .}}"], text=True)

        data = _json.loads(out.strip())

        return {

            "disk_used": data.get("DiskUsed"),

            "disk_total": data.get("DiskTotal"),

            "disk_used_percent": data.get("DiskUsedPercent"),

        }

    except Exception:

        return None



def snapshot_wsl_vhd():

    vhdx = os.path.join(os.environ.get("LOCALAPPDATA", ""), "Docker", "wsl", "data", "ext4.vhdx")

    if not os.path.exists(vhdx):

        return None

    size_gb = os.path.getsize(vhdx) / (1024**3)

    return {"path": vhdx, "size_gb": round(size_gb, 1)}

def load_plugins(parser):

    for path in glob.glob(os.path.join(PLUGINS_DIR, "*.py")):

        name = os.path.splitext(os.path.basename(path))[0]

        spec = importlib.util.spec_from_file_location(name, path)

        mod = importlib.util.module_from_spec(spec)

        try:

            spec.loader.exec_module(mod)

            if hasattr(mod, "register"):

                mod.register(parser, emit_command, ROOT, BUS)

        except Exception as e:

            print(f"Plugin {name} failed: {e}")

def emit_event(event, level="info", source="model_keeper", meta=None):

    payload = {

        "type": "event",

        "source": source,

        "event": event,

        "level": level,

        "timestamp": datetime.now(timezone.utc).isoformat(),

        "meta": meta or {}

    }

    fname = f"event_{datetime.now().strftime('%Y%m%d_%H%M%S_%f')}.json"

    path = os.path.join(EVENTS, fname)

    with open(path, "w", encoding="utf-8") as f:

        json.dump(payload, f, indent=2)



def gpu_available():

    try:

        import torch

        return torch.cuda.is_available()

    except Exception:

        return False



def free_gpu_memory_gb():

    try:

        import torch

        if not torch.cuda.is_available():

            return 0.0

        total = torch.cuda.get_device_properties(0).total_memory

        reserved = torch.cuda.memory_reserved(0)

        allocated = torch.cuda.memory_allocated(0)

        free = total - max(reserved, allocated)

        return free / (1024**3)

    except Exception:

        return 0.0



def estimate_model_size_gb(model_path_or_id):

    # crude heuristic: assume 8–16 GB; user can refine later

    return 8.0



def choose_device(model):

    dev = model.get("device", "auto")

    if dev in ["cpu", "cuda"]:

        return dev

    # auto mode

    if gpu_available():

        free_gb = free_gpu_memory_gb()

        est_gb = estimate_model_size_gb(model["model"])

        if est_gb <= free_gb * 0.8:

            return "cuda"

    return "cpu"



def load_endpoints():

    if not os.path.exists(ENDPOINTS_PATH):

        return []

    with open(ENDPOINTS_PATH, "r", encoding="utf-8") as f:

        data = json.load(f)

    return data.get("endpoints", [])



def start_endpoint(ep):

    name = ep["name"]

    port = ep["port"]

    model = ep["model"]

    device = choose_device(ep)



    cmd = [

        "python",

        "-m", "vllm.entrypoints.openai.api_server",

        "--model", model,

        "--port", str(port),

        "--host", "127.0.0.1",

        "--device", device

    ]



    emit_event("endpoint_starting", meta={"name": name, "port": port, "model": model, "device": device})

    proc = subprocess.Popen(cmd, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)

    return proc, device



def main():

    emit_event("model_keeper_started")

    procs = {}



    while True:

        # heartbeat

        emit_event("heartbeat", meta={"endpoints": list(procs.keys())})



        endpoints = load_endpoints()

        desired = {ep["name"]: ep for ep in endpoints if ep.get("persistent", True)}



        # start missing

        for name, ep in desired.items():

            if name not in procs or procs[name]["proc"].poll() is not None:

                proc, device = start_endpoint(ep)

                procs[name] = {"proc": proc, "device": device, "port": ep["port"], "model": ep["model"]}



        # stop extra

        for name in list(procs.keys()):

            if name not in desired:

                emit_event("endpoint_stopping", meta={"name": name})

                try:

                    procs[name]["proc"].terminate()

                except Exception:

                    pass

                del procs[name]



        time.sleep(10)



if __name__ == "__main__":

    main()

@app.route("/api/status")

def api_status():

    now = datetime.now(timezone.utc)

    latest = {}



    files = sorted(os.listdir(EVENTS), reverse=True)

    for f in files:

        try:

            with open(os.path.join(EVENTS, f), "r", encoding="utf-8") as fp:

                data = json.load(fp)

            if data.get("event") == "heartbeat":

                src = data.get("source")

                ts = datetime.fromisoformat(data["timestamp"].replace("Z","+00:00"))

                if src not in latest or ts > latest[src]:

                    latest[src] = ts

        except:

            continue



    status = {}

    for agent, ts in latest.items():

        age = (now - ts).total_seconds()

        if age < 30:

            state = "ONLINE"

        elif age < 120:

            state = "LAGGING"

        else:

            state = "STALE"

        status[agent] = {"state": state, "age": age}



    return jsonify(status)

import os

# Utility: Emit command to bus

def emit_command(target, command, args=None):

    args = args or {}

    payload = {

        "type": "command",

        "target": target,

        "command": command,

        "args": args,

        "issued_by": "cli",

        "timestamp": datetime.now(timezone.utc).isoformat()

    }

    fname = f"cmd_{datetime.now().strftime('%Y%m%d_%H%M%S_%f')}.json"

    path = os.path.join(COMMANDS, fname)

    with open(path, "w", encoding="utf-8") as f:

        json.dump(payload, f, indent=2)

    print(f"✔ Command sent → {target}:{command}")



# ------------------------------------------------------------

# Event tailing

# ------------------------------------------------------------

def tail_events():

    print("=== Tailing BossCrafts Events ===")

    seen = set()

    while True:

        files = sorted(os.listdir(EVENTS))

        for f in files:

            if f not in seen:

                seen.add(f)

                try:

                    with open(os.path.join(EVENTS, f), "r", encoding="utf-8") as fp:

                        data = json.load(fp)

                    ts = data.get("timestamp")

                    src = data.get("source")

                    evt = data.get("event")

                    lvl = data.get("level")

                    act = data.get("action")

                    print(f"[{ts}] {lvl.upper():8} {src:14} → {evt} ({act})")

                except:

                    pass

        time.sleep(1)



# ------------------------------------------------------------

# Agent status

# ------------------------------------------------------------

def show_status():

    now = datetime.now(timezone.utc)

    latest = {}



    files = sorted(os.listdir(EVENTS), reverse=True)

    for f in files:

        try:

            with open(os.path.join(EVENTS, f), "r", encoding="utf-8") as fp:

                data = json.load(fp)

            if data.get("event") == "heartbeat":

                src = data.get("source")

                ts = datetime.fromisoformat(data["timestamp"].replace("Z","+00:00"))

                if src not in latest or ts > latest[src]:

                    latest[src] = ts

        except:

            continue



    print("=== BossCrafts Agent Status ===")

    for agent, ts in latest.items():

        age = (now - ts).total_seconds()

        if age < 30:

            state = "ONLINE"

        elif age < 120:

            state = "LAGGING"

        else:

            state = "STALE"

        print(f"{agent:14} → {state:7} ({int(age)}s ago)")

def llm_list():

    if not os.path.exists(ENDPOINTS_PATH):

        print("No endpoints.json found.")

        return

    with open(ENDPOINTS_PATH, "r", encoding="utf-8") as f:

        cfg = json.load(f)

    print("=== Model Endpoints ===")

    for ep in cfg.get("endpoints", []):

        print(f"- {ep['name']} @ {ep['port']} ({ep['model']}, device={ep['device']})")



def llm_query(prompt, endpoint="main"):

    if not os.path.exists(ENDPOINTS_PATH):

        print("No endpoints.json found.")

        return

    with open(ENDPOINTS_PATH, "r", encoding="utf-8") as f:

        cfg = json.load(f)

    eps = {ep["name"]: ep for ep in cfg.get("endpoints", [])}

    if endpoint not in eps:

        print(f"No such endpoint: {endpoint}")

        return

    url = f"http://127.0.0.1:{eps[endpoint]['port']}/v1/completions"

    payload = {"prompt": prompt, "max_tokens": 256}

    r = requests.post(url, json=payload)

    print(r.json())

def ritual_record(name):

    path = os.path.join(RITUALS, f"{name}.json")

    print(f"Recording ritual '{name}'. Type commands. 'done' to finish.")

    steps = []

    while True:

        line = input("ritual> ").strip()

        if line in ["done", "exit", "quit"]:

            break

        if line:

            steps.append(line)

    with open(path, "w", encoding="utf-8") as f:

        json.dump({"name": name, "steps": steps}, f, indent=2)

    print(f"✔ Ritual saved: {path}")



def ritual_play(name):

    path = os.path.join(RITUALS, f"{name}.json")

    if not os.path.exists(path):

        print(f"No ritual named '{name}'.")

        return

    with open(path, "r", encoding="utf-8") as f:

        data = json.load(f)

    print(f"Playing ritual '{name}'...")

    for line in data.get("steps", []):

        print(f"→ {line}")

        args = build_parser().parse_args(line.split())

        if hasattr(args, "func"):

            args.func(args)

        time.sleep(0.2)



def ritual_list():

    files = [f for f in os.listdir(RITUALS) if f.endswith(".json")]

def shell():

# Plugin loader

def load_plugins(parser):

    for fname in os.listdir(PLUGINS):

        if fname.endswith(".py"):

            path = os.path.join(PLUGINS, fname)

            try:

                import importlib.util

                spec = importlib.util.spec_from_file_location(fname[:-3], path)

                mod = importlib.util.module_from_spec(spec)

                spec.loader.exec_module(mod)

                if hasattr(mod, "register"):

                    mod.register(parser, emit_command, ROOT, BUS)

            except Exception as e:

                print(f"Plugin {fname} failed: {e}")



# ------------------------------------------------------------

# CLI parser

# ------------------------------------------------------------

def build_parser():

    p = argparse.ArgumentParser(description="BossCrafts CLI")

    sub = p.add_subparsers(dest="cmd")



    # status

    s = sub.add_parser("status")

    s.set_defaults(func=lambda a: show_status())



    # tail

    s = sub.add_parser("tail")

    s.set_defaults(func=lambda a: tail_events())



    # agent

    s = sub.add_parser("agent")

    s.add_argument("agent")

    s.add_argument("command")

    s.add_argument("--args", default="{}")

    def agent_cmd(a):

        if a.agent not in AGENTS:

            print(f"Unknown agent: {a.agent}")

            return

        try:

            args = json.loads(a.args)

        except:

            args = {"raw": a.args}

        emit_command(AGENTS[a.agent], a.command, args)

    s.set_defaults(func=agent_cmd)



    # llm

    s = sub.add_parser("llm")

    ss = s.add_subparsers(dest="sub")



    s1 = ss.add_parser("list")

    s1.set_defaults(func=lambda a: llm_list())



    s2 = ss.add_parser("query")

    s2.add_argument("prompt")

    s2.add_argument("--endpoint", default="main")

    s2.set_defaults(func=lambda a: llm_query(a.prompt, a.endpoint))



    # rituals

    s = sub.add_parser("ritual")

    ss = s.add_subparsers(dest="sub")



    s1 = ss.add_parser("record")

    s1.add_argument("name")

    s1.set_defaults(func=lambda a: ritual_record(a.name))



    s2 = ss.add_parser("play")

    s2.add_argument("name")

    s2.set_defaults(func=lambda a: ritual_play(a.name))



    s3 = ss.add_parser("list")

    s3.set_defaults(func=lambda a: ritual_list())



    # shell

    s = sub.add_parser("shell")

    s.set_defaults(func=lambda a: shell())



    return p



# ------------------------------------------------------------

# Main

# ------------------------------------------------------------

def main():

    parser = build_parser()

    load_plugins(parser)

    args = parser.parse_args()

    if hasattr(args, "func"):

        args.func(args)

    else:

        parser.print_help()



if __name__ == "__main__":

    main()
